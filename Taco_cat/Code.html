                 function TacoCat() {
    //Get text input and create strings for palindrome text
    let TcWord = document.getElementById("InputT").value;
    let Word = TcWord.toLowerCase();
    Word = Word.replace(/\s/g, "");
    Word = Word.replace(/\W/g, "");
    RevWord = Word.split("").reverse().join("");
    
    //Palindrome Check
    if (RevWord == Word) {
        document.getElementById("OutputT"). innerText = `${TcWord} becomes ${Word}. The reverse of ${Word} is ${RevWord}, therefore ${TcWord} is a palindrome`;
    }
    else {
        document.getElementById("OutputT").innerText = `${TcWord} becomes ${Word}. The reverse of ${Word} is ${RevWord}, therefore ${TcWord} is not a palindrome`;
    }
    
    
    }
